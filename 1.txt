-- Create the Employee table with the DepartmentID column
CREATE TABLE Employee (
    EmployeeID INT PRIMARY KEY,
    FirstName VARCHAR(50),
    LastName VARCHAR(50),
    Salary DECIMAL(10, 2),
    HireDate DATE,
    DepartmentID INT -- This is the new column
